function generateRandomLayout(numbers) { const layout = []; const boardWidth = 430; const boardHeight = 520; const gap = 8; numbers.forEach(num => { let placed = false; let attempts = 0; while (!placed && attempts < 1000) { attempts++; const size = Math.floor(Math.random() * 24) + 42; const fontSize = Math.floor(size / 2.4); const x = Math.random() * (boardWidth - size); const y = Math.random() * (boardHeight - size); const newItem = { number: num, width: size, height: size, fontSize: fontSize, leftPx: x, topPx: y, rotate: Math.floor(Math.random() * 31) - 15 }; const isOverlapping = layout.some(item => { return !( newItem.leftPx + newItem.width + gap < item.leftPx || newItem.leftPx > item.leftPx + item.width + gap || newItem.topPx + newItem.height + gap < item.topPx || newItem.topPx > item.topPx + item.height + gap ); }); if (!isOverlapping) { layout.push(newItem); placed = true; } } }); return layout; }
